## การตั้งค่า Database (PostgreSQL)

ก่อนเริ่มรันโปรเจกต์ จำเป็นต้องจำลองฐานข้อมูลในเครื่องของคุณตามขั้นตอนดังนี้:

### 1.สร้าง User และ Database (ผ่าน pgAdmin 4)
1. เปิดโปรแกรม pgAdmin 4
2. สร้าง User ใหม่:
   - Name: `foodhubuser`
   - Password: `FoodHubPass123`
   - *หมายเหตุ: ต้องให้สิทธิ์ Can login และ Superuser หรือสิทธิ์ในการสร้าง DB แก่ User นี้*
3. สร้าง Database ใหม่:
   - Database Name: `FoodHubDb`
   - Owner: เลือกเป็น `foodhubuser`

### 2.รัน SQL Scripts เพื่อสร้างโครงสร้างและข้อมูล
1. คลิกขวาที่ `FoodHubDb` เลือก **Query Tool**
2. เลือก Connection ให้ใช้ User `foodhubuser` (สำคัญมาก เพื่อป้องกัน Error Permission Denied)
3. **รัน Script ส่วนที่ 1 (สร้างตาราง):** คัดลอกโค้ดด้านล่างไปวางแล้วกด Execute

ส่วนที่ 1

```sql
DROP TABLE IF EXISTS "Reviews" CASCADE;
DROP TABLE IF EXISTS "Promotions" CASCADE;
DROP TABLE IF EXISTS "Bookings" CASCADE;
DROP TABLE IF EXISTS "Restaurants" CASCADE;
DROP TABLE IF EXISTS "Customers" CASCADE;

CREATE TABLE "Customers" (
    "CustomerId" integer GENERATED BY DEFAULT AS IDENTITY,
    "Name" character varying(100) NOT NULL,
    "Email" character varying(150) NOT NULL,
    "Phone" character varying(20),
    "CreatedAt" timestamp with time zone DEFAULT CURRENT_TIMESTAMP NOT NULL,
    "LateCancelCount" integer DEFAULT 0 NOT NULL,
    CONSTRAINT "PK_Customers" PRIMARY KEY ("CustomerId"),
    CONSTRAINT "UQ_Customers_Email" UNIQUE ("Email")
);

CREATE TABLE "Restaurants" (
    "RestaurantId" integer GENERATED BY DEFAULT AS IDENTITY,
    "Name" character varying(130) NOT NULL,
    "OwnerName" character varying(100) NOT NULL,
    "Email" character varying(150) NOT NULL,
    "Phone" character varying(20),
    "Address" text,
    "Category" character varying(50),
    "OpenHours" character varying(100),
    "AvgRating" double precision DEFAULT 0.0,
    "CreatedAt" timestamp with time zone DEFAULT CURRENT_TIMESTAMP NOT NULL,
    "MaxCapacity" integer DEFAULT 20 NOT NULL,
    CONSTRAINT "PK_Restaurants" PRIMARY KEY ("RestaurantId"),
    CONSTRAINT "UQ_Restaurants_Email" UNIQUE ("Email")
);

CREATE TABLE "Bookings" (
    "BookingId" integer GENERATED BY DEFAULT AS IDENTITY,
    "CustomerId" integer NOT NULL,
    "RestaurantId" integer NOT NULL,
    "BookingDate" timestamp with time zone NOT NULL,
    "NumPeople" integer NOT NULL,
    "Status" character varying(20) NOT NULL,
    "CreatedAt" timestamp with time zone DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT "PK_Bookings" PRIMARY KEY ("BookingId"),
    CONSTRAINT "FK_Bookings_Customers_CustomerId" FOREIGN KEY ("CustomerId") REFERENCES "Customers" ("CustomerId") ON DELETE CASCADE,
    CONSTRAINT "FK_Bookings_Restaurants_RestaurantId" FOREIGN KEY ("RestaurantId") REFERENCES "Restaurants" ("RestaurantId") ON DELETE CASCADE,
    CONSTRAINT "CK_Bookings_NumPeople" CHECK ("NumPeople" > 0)
);

CREATE TABLE "Reviews" (
    "ReviewId" integer GENERATED BY DEFAULT AS IDENTITY,
    "CustomerId" integer NOT NULL,
    "RestaurantId" integer NOT NULL,
    "BookingId" integer NULL,
    "Rating" integer NOT NULL,
    "Comment" text,
    "CreatedAt" timestamp with time zone DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT "PK_Reviews" PRIMARY KEY ("ReviewId"),
    CONSTRAINT "FK_Reviews_Customers_CustomerId" FOREIGN KEY ("CustomerId") REFERENCES "Customers" ("CustomerId") ON DELETE CASCADE,
    CONSTRAINT "FK_Reviews_Restaurants_RestaurantId" FOREIGN KEY ("RestaurantId") REFERENCES "Restaurants" ("RestaurantId") ON DELETE CASCADE,
    CONSTRAINT "FK_Reviews_Bookings_BookingId" FOREIGN KEY ("BookingId") REFERENCES "Bookings" ("BookingId") ON DELETE SET NULL,
    CONSTRAINT "CK_Reviews_Rating" CHECK ("Rating" BETWEEN 1 AND 5)
);

CREATE TABLE "Promotions" (
    "PromotionId" integer GENERATED BY DEFAULT AS IDENTITY,
    "RestaurantId" integer NOT NULL,
    "Title" character varying(150) NOT NULL,
    "Description" text,
    "Discount" real NOT NULL,
    "ExpireDate" timestamp with time zone NOT NULL,
    "IsActive" boolean DEFAULT true NOT NULL,
    CONSTRAINT "PK_Promotions" PRIMARY KEY ("PromotionId"),
    CONSTRAINT "FK_Promotions_Restaurants_RestaurantId" FOREIGN KEY ("RestaurantId") REFERENCES "Restaurants" ("RestaurantId") ON DELETE CASCADE
);

CREATE INDEX "IX_Bookings_CustomerId" ON "Bookings" ("CustomerId");
CREATE INDEX "IX_Bookings_RestaurantId" ON "Bookings" ("RestaurantId");
CREATE INDEX "IX_Promotions_RestaurantId" ON "Promotions" ("RestaurantId");
CREATE INDEX "IX_Reviews_BookingId" ON "Reviews" ("BookingId");
CREATE INDEX "IX_Reviews_CustomerId" ON "Reviews" ("CustomerId");
CREATE INDEX "IX_Reviews_RestaurantId" ON "Reviews" ("RestaurantId");
```

## วิธีการรันโปรเจกต์
เมื่อตั้งค่าฐานข้อมูลเสร็จแล้ว ให้รันโปรเจกต์ตามลำดับต่อไปนี้:
1. รัน Server API: เข้าไปที่โฟลเดอร์ FoodHub เปิด Solution ด้วย Visual Studio แล้วกดรัน (F5) ลองทดสอบยิง API ผ่าน Swagger UI ดูว่าดึงข้อมูลได้ปกติหรือไม่
2. รัน Client Apps: เมื่อ Server รันอยู่ (ห้ามปิด) ให้เปิดอีกหน้าต่าง Visual Studio สำหรับโปรเจกต์ FoodHubCustomerUI หรือ FoodHubRestaurantUI แล้วกดรันเพื่อทดสอบการใช้งานจริงผ่านหน้าแอป
